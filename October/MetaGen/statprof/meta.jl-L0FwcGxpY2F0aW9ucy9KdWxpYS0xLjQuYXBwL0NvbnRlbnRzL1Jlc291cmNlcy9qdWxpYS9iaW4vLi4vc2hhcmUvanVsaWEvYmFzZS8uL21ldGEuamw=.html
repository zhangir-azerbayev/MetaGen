152 """
153 parse(str, start; greedy=true, raise=true, depwarn=true)
154
155 Parse the expression string and return an expression (which could later be passed to eval
156 for execution). `start` is the index of the first character to start parsing. If `greedy` is
157 `true` (default), `parse` will try to consume as much input as it can; otherwise, it will
158 stop as soon as it has parsed a valid expression. Incomplete but otherwise syntactically
159 valid expressions will return `Expr(:incomplete, "(error message)")`. If `raise` is `true`
160 (default), syntax errors other than incomplete expressions will raise an error. If `raise`
161 is `false`, `parse` will return an expression that will raise an error upon evaluation. If
162 `depwarn` is `false`, deprecation warnings will be suppressed.
163
164 ```jldoctest
165 julia> Meta.parse("x = 3, y = 5", 7)
166 (:(y = 5), 13)
167
168 julia> Meta.parse("x = 3, y = 5", 5)
169 (:((3, y) = 5), 13)
170 ```

172 function parse(str::AbstractString, pos::Integer; greedy::Bool=true, raise::Bool=true,
173 depwarn::Bool=true)
174 # pos is one based byte offset.
175 # returns (expr, end_pos). expr is () in case of parse error.
176 bstr = String(str)
177 # For now, assume all parser warnings are depwarns
178 ex, pos = with_logger(depwarn ? current_logger() : NullLogger()) do
179 ccall(:jl_parse_string, Any,
180 (Ptr{UInt8}, Csize_t, Int32, Int32),
181 bstr, sizeof(bstr), pos-1, greedy ? 1 : 0)
182 end
183 if raise && isa(ex,Expr) && ex.head === :error
184 throw(ParseError(ex.args[1]))
185 end
186 return ex, pos+1 # C is zero-based, Julia is 1-based
187 end

189 """
190 parse(str; raise=true, depwarn=true)
191
192 Parse the expression string greedily, returning a single expression. An error is thrown if
193 there are additional characters after the first expression. If `raise` is `true` (default),
194 syntax errors will raise an error; otherwise, `parse` will return an expression that will
195 raise an error upon evaluation. If `depwarn` is `false`, deprecation warnings will be
196 suppressed.
197
198 ```jldoctest
199 julia> Meta.parse("x = 3")
200 :(x = 3)
201
202 julia> Meta.parse("x = ")
203 :($(Expr(:incomplete, "incomplete: premature end of input")))
204
205 julia> Meta.parse("1.0.2")
206 ERROR: Base.Meta.ParseError("invalid numeric constant \\\"1.0.\\\"")
207 Stacktrace:
208 [...]
209
210 julia> Meta.parse("1.0.2"; raise = false)
211 :($(Expr(:error, "invalid numeric constant \"1.0.\"")))
212 ```
213 """
214 function parse(str::AbstractString; raise::Bool=true, depwarn::Bool=true)
215 ex, pos = parse(str, 1, greedy=true, raise=raise, depwarn=depwarn)
216 if isa(ex,Expr) && ex.head === :error
217 return ex
218 end
219 if pos <= ncodeunits(str)
220 raise && throw(ParseError("extra token after end of expression"))
221 return Expr(:error, "extra token after end of expression")
222 end
223 return ex
224 end

241

248 - turn off boundschecking (if `boundscheck === :off`) or propagate boundschecking (if `boundscheck === :propagate`)

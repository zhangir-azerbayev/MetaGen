78 sprint(f::Function, args...; context=nothing, sizehint=0)
79
80 Call the given function with an I/O stream and the supplied extra arguments.
81 Everything written to this I/O stream is returned as a string.
82 `context` can be either an [`IOContext`](@ref) whose properties will be used,
83 or a `Pair` specifying a property and its value. `sizehint` suggests the capacity
84 of the buffer (in bytes).
85
86 The optional keyword argument `context` can be set to `:key=>value` pair
87 or an `IO` or [`IOContext`](@ref) object whose attributes are used for the I/O
88 stream passed to `f`. The optional `sizehint` is a suggested size (in bytes)
89 to allocate for the buffer used to write the string.
90
91 # Examples
92 ```jldoctest
93 julia> sprint(show, 66.66666; context=:compact => true)
94 "66.6667"
95
96 julia> sprint(showerror, BoundsError([1], 100))
97 "BoundsError: attempt to access 1-element Array{Int64,1} at index [100]"
98 ```
99 """
100 function sprint(f::Function, args...; context=nothing, sizehint::Integer=0)
101 s = IOBuffer(sizehint=sizehint)
102 if context !== nothing
103 f(IOContext(s, context), args...)
104 else
105 f(s, args...)
106 end
107 String(resize!(s.data, s.size))
108 end

254 """
255 join([io::IO,] strings [, delim [, last]])
256
257 Join an array of `strings` into a single string, inserting the given delimiter (if any) between
258 adjacent strings. If `last` is given, it will be used instead of `delim` between the last
259 two strings. If `io` is given, the result is written to `io` rather than returned as
260 as a `String`.
261
262 `strings` can be any iterable over elements `x` which are convertible to strings
263 via `print(io::IOBuffer, x)`. `strings` will be printed to `io`.
264
265 # Examples
266 ```jldoctest
267 julia> join(["apples", "bananas", "pineapples"], ", ", " and ")
268 "apples, bananas and pineapples"
269
270 julia> join([1,2,3,4,5])
271 "12345"
272 ```
273 """
274 function join(io::IO, strings, delim, last)
275 first = true
276 local prev
277 for str in strings
278 if @isdefined prev
279 first ? (first = false) : print(io, delim)
280 print(io, prev)
281 end
282 prev = str
283 end
284 if @isdefined prev
285 first || print(io, last)
286 print(io, prev)
287 end
288 nothing
289 end

310 """
311 escape_string(str::AbstractString[, esc])::AbstractString
312 escape_string(io, str::AbstractString[, esc::])::Nothing
313
314 General escaping of traditional C and Unicode escape sequences. The first form returns the
315 escaped string, the second prints the result to `io`.
316
317 Backslashes (`\\`) are escaped with a double-backslash (`"\\\\"`). Non-printable
318 characters are escaped either with their standard C escape codes, `"\\0"` for NUL (if
319 unambiguous), unicode code point (`"\\u"` prefix) or hex (`"\\x"` prefix).
320
321 The optional `esc` argument specifies any additional characters that should also be
322 escaped by a prepending backslash (`\"` is also escaped by default in the first form).
323
324 # Examples
325 ```jldoctest
326 julia> escape_string("aaa\\nbbb")
327 "aaa\\\\nbbb"
328
329 julia> escape_string("\\xfe\\xff") # invalid utf-8
330 "\\\\xfe\\\\xff"
331
332 julia> escape_string(string('\\u2135','\\0')) # unambiguous
333 "\\\\0"
334
335 julia> escape_string(string('\\u2135','\\0','0')) # \\0 would be ambiguous
336 "\\\\x000"
337 ```
338
339 ## See also
340 [`unescape_string`](@ref) for the reverse operation.
341 """
342 function escape_string(io::IO, s::AbstractString, esc="")
343 a = Iterators.Stateful(s)
344 for c in a
345 if c in esc
346 print(io, '\\', c)
347 elseif isascii(c)
348 c == '\0' ? print(io, escape_nul(peek(a))) :
349 c == '\e' ? print(io, "\\e") :
350 c == '\\' ? print(io, "\\\\") :
351 '\a' <= c <= '\r' ? print(io, '\\', "abtnvfr"[Int(c)-6]) :
352 isprint(c) ? print(io, c) :
353 print(io, "\\x", string(UInt32(c), base = 16, pad = 2))
354 elseif !isoverlong(c) && !ismalformed(c)
355 isprint(c) ? print(io, c) :
356 c <= '\x7f' ? print(io, "\\x", string(UInt32(c), base = 16, pad = 2)) :
357 c <= '\uffff' ? print(io, "\\u", string(UInt32(c), base = 16, pad = need_full_hex(peek(a)) ? 4 : 2)) :
358 print(io, "\\U", string(UInt32(c), base = 16, pad = need_full_hex(peek(a)) ? 8 : 4))
359 else # malformed or overlong
360 u = bswap(reinterpret(UInt32, c))
361 while true
362 print(io, "\\x", string(u % UInt8, base = 16, pad = 2))
363 (u >>= 8) == 0 && break
364 end
365 end
366 end
367 end

380 """
381 unescape_string(str::AbstractString, keep = ())::AbstractString
382 unescape_string(io, s::AbstractString, keep = ())::Nothing
383
384 General unescaping of traditional C and Unicode escape sequences. The first form returns
385 the escaped string, the second prints the result to `io`.
386 The argument `keep` specifies a collection of characters which (along with backlashes) are
387 to be kept as they are.
388
389 The following escape sequences are recognised:
390 - Escaped backslash (`\\\\`)
391 - Escaped double-quote (`\\\"`)
392 - Standard C escape sequences (`\\a`, `\\b`, `\\t`, `\\n`, `\\v`, `\\f`, `\\r`, `\\e`)
393 - Unicode BMP code points (`\\u` with 1-4 trailing hex digits)
394 - All Unicode code points (`\\U` with 1-8 trailing hex digits; max value = 0010ffff)
395 - Hex bytes (`\\x` with 1-2 trailing hex digits)
396 - Octal bytes (`\\` with 1-3 trailing octal digits)
397
398 # Examples
399 ```jldoctest
400 julia> unescape_string("aaa\\\\nbbb") # C escape sequence
401 "aaa\\nbbb"
402
403 julia> unescape_string("\\\\u03c0") # unicode
404 ""
405
406 julia> unescape_string("\\\\101") # octal
407 "A"
408
409 julia> unescape_string("aaa \\\\g \\\\n", ['g']) # using `keep` argument
410 "aaa \\\\g \\n"
411 ```
412
413 ## See also
414 [`escape_string`](@ref).
415 """
416 function unescape_string(io::IO, s::AbstractString, keep = ())
417 a = Iterators.Stateful(s)
418 for c in a
419 if !isempty(a) && c == '\\'
420 c = popfirst!(a)
421 if c in keep
422 print(io, '\\', c)
423 elseif c == 'x' || c == 'u' || c == 'U'
424 n = k = 0
425 m = c == 'x' ? 2 :
426 c == 'u' ? 4 : 8
427 while (k += 1) <= m && !isempty(a)
428 nc = peek(a)
429 n = '0' <= nc <= '9' ? n<<4 + (nc-'0') :
430 'a' <= nc <= 'f' ? n<<4 + (nc-'a'+10) :
431 'A' <= nc <= 'F' ? n<<4 + (nc-'A'+10) : break
432 popfirst!(a)
433 end
434 if k == 1 || n > 0x10ffff
435 u = m == 4 ? 'u' : 'U'
436 throw(ArgumentError("invalid $(m == 2 ? "hex (\\x)" :
437 "unicode (\\$u)") escape sequence"))
438 end
439 if m == 2 # \x escape sequence
440 write(io, UInt8(n))
441 else
442 print(io, Char(n))
443 end
444 elseif '0' <= c <= '7'
445 k = 1
446 n = c-'0'
447 while (k += 1) <= 3 && !isempty(a)
448 c = peek(a)
449 n = ('0' <= c <= '7') ? n<<3 + c-'0' : break
450 popfirst!(a)
451 end
452 if n > 255
453 throw(ArgumentError("octal escape sequence out of range"))
454 end
455 write(io, UInt8(n))
456 else
457 print(io, c == 'a' ? '\a' :
458 c == 'b' ? '\b' :
459 c == 't' ? '\t' :
460 c == 'n' ? '\n' :
461 c == 'v' ? '\v' :
462 c == 'f' ? '\f' :
463 c == 'r' ? '\r' :
464 c == 'e' ? '\e' :
465 (c == '\\' || c == '"') ? c :
466 throw(ArgumentError("invalid escape sequence \\$c")))
467 end
468 else
469 print(io, c)
470 end
471 end

566 ```jldoctest

574 function unindent(str::AbstractString, indent::Int; tabwidth=8)
575 indent == 0 && return str
576 # Note: this loses the type of the original string
577 buf = IOBuffer(sizehint=sizeof(str))
578 cutting = true
579 col = 0 # current column (0 based)
580 for ch in str
581 if cutting
582 if ch == ' '
583 col += 1
584 elseif ch == '\t'
585 col = div(col + tabwidth, tabwidth) * tabwidth
586 elseif ch == '\n'
587 # Now we need to output enough indentation
588 for i = 1:col-indent
589 print(buf, ' ')
590 end
591 col = 0
592 print(buf, '\n')
593 else
594 cutting = false
595 # Now we need to output enough indentation to get to
596 # correct place
597 for i = 1:col-indent
598 print(buf, ' ')
599 end
600 col += 1
601 print(buf, ch)
602 end
603 elseif ch == '\t' # Handle internal tabs
604 upd = div(col + tabwidth, tabwidth) * tabwidth
605 # output the number of spaces that would have been seen
606 # with original indentation
607 for i = 1:(upd-col)
608 print(buf, ' ')
609 end
610 col = upd
611 elseif ch == '\n'
612 cutting = true
613 col = 0
614 print(buf, '\n')
615 else
616 col += 1
617 print(buf, ch)
618 end
619 end
620 # If we were still "cutting" when we hit the end of the string,
621 # we need to output the right number of spaces for the indentation
622 if cutting
623 for i = 1:col-indent
624 print(buf, ' ')
625 end
626 end
627 String(take!(buf))
628 end

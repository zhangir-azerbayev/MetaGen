184
185 Validate `mi`, logging any violation by pushing an `InvalidCodeError` into `errors`.
186
187 If `isa(c, CodeInfo)`, also call `validate_code!(errors, c)`. It is assumed that `c` is
188 the `CodeInfo` instance associated with `mi`.
189 """
190 function validate_code!(errors::Vector{>:InvalidCodeError}, mi::Core.MethodInstance,
191 c::Union{Nothing,CodeInfo} = Core.Compiler.retrieve_code_info(mi))
192 is_top_level = mi.def isa Module
193 if is_top_level
194 mnargs = 0
195 else
196 m = mi.def::Method
197 mnargs = m.nargs
198 n_sig_params = length(Core.Compiler.unwrap_unionall(m.sig).parameters)
199 if (m.isva ? (n_sig_params < (mnargs - 1)) : (n_sig_params != mnargs))
200 push!(errors, InvalidCodeError(SIGNATURE_NARGS_MISMATCH, (m.isva, n_sig_params, mnargs)))
201 end
202 end
203 if isa(c, CodeInfo)
204 mnargs > length(c.slotnames) && push!(errors, InvalidCodeError(SLOTNAMES_NARGS_MISMATCH))
205 validate_code!(errors, c, is_top_level)
206 end
207 return errors
208 end

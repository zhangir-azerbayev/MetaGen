249 truncate :: Union{Bool,Nothing} = nothing,
250 append :: Union{Bool,Nothing} = nothing,
251 )
252 if write === true && read !== true && append !== true
253 create === nothing && (create = true)
254 truncate === nothing && (truncate = true)

259 create === nothing && (create = true)

747 # requires that indices for target are the integer unit range from firstindex to lastindex
748 # returns whether the delimiter was matched
749 # uses the KnuthMorrisPratt_algorithm, with the first and second cache entries unrolled
750 # For longer targets, the cache improves the big-O efficiency of scanning of sequences
751 # with repeated patterns
752 # Each cache entry tells us which index we should start the search at.
753 # We assume this is unlikely, so we only lazy-initialize as much of the cache as we need to use
754 # When we allocate the cache, we initialize it to 0 (and offset by the first index afterwards).
755 # Suppose target is:
756 # Index: 1245689
757 # Value: "acacx"
758 # We would set the cache to
759 # 0 0 0 1 2 3 4 0
760 # So after if we mismatch after the second ac sequence,
761 # we can immediately jump back to index 5 (4 + 1).
762 function readuntil_vector!(io::IO, target::AbstractVector{T}, keep::Bool, out) where {T}
763 first = firstindex(target)
764 last = lastindex(target)
765 len = last - first + 1
766 if len < 1
767 return true
768 end
769 pos = 0 # array-offset
770 max_pos = 1 # array-offset in cache
771 local cache # will be lazy initialized when needed
772 output! = (isa(out, IO) ? write : push!)
773 while !eof(io)
774 c = read(io, T)
775 # Backtrack until the next target character matches what was found
776 while true
777 c1 = target[pos + first]
778 if c == c1
779 pos += 1
780 break
781 elseif pos == 0
782 break
783 elseif pos == 1
784 if !keep
785 output!(out, target[first])
786 end
787 pos = 0
788 else
789 # grow cache to contain up to `pos`
790 if !@isdefined(cache)
791 cache = zeros(Int, len)
792 end
793 while max_pos < pos
794 ci = target[max_pos + first]
795 b = max_pos
796 max_pos += 1
797 while b != 0
798 b = cache[b]
799 cb = target[b + first]
800 if ci == cb
801 cache[max_pos] = b + 1
802 break
803 end
804 end
805 end
806 # read new position from cache
807 pos1 = cache[pos]
808 if !keep
809 # and add the removed prefix from the target to the output
810 # if not always keeping the match
811 for b in 1:(pos - pos1)
812 output!(out, target[b - 1 + first])
813 end
814 end
815 pos = pos1
816 end
817 end
818 if keep || pos == 0
819 output!(out, c)
820 end
821 pos == len && return true
822 end
823 if !keep
824 # failed early without finishing the match,
825 # add the partial match to the output
826 # if not always keeping the match
827 for b in 1:pos
828 output!(out, target[b - 1 + first])
829 end
830 end
831 return false
832 end

834 function readuntil(io::IO, target::AbstractString; keep::Bool=false)
835 # small-string target optimizations
836 isempty(target) && return ""
837 c, rest = Iterators.peel(target)
838 if isempty(rest) && c <= '\x7f'
839 return readuntil_string(io, c % UInt8, keep)
840 end
841 # convert String to a utf8-byte-iterator
842 if !(target isa String) && !(target isa SubString{String})
843 target = String(target)
844 end
845 target = codeunits(target)::AbstractVector
846 return String(readuntil(io, target, keep=keep))
847 end

876
877 """
878 readbytes!(stream::IO, b::AbstractVector{UInt8}, nb=length(b))
879
880 Read at most `nb` bytes from `stream` into `b`, returning the number of bytes read.
881 The size of `b` will be increased if needed (i.e. if `nb` is greater than `length(b)`
882 and enough bytes could be read), but it will never be decreased.
883 """
884 function readbytes!(s::IO, b::AbstractArray{UInt8}, nb=length(b))
885 require_one_based_indexing(b)
886 olb = lb = length(b)
887 nr = 0
888 while nr < nb && !eof(s)
889 a = read(s, UInt8)
890 nr += 1
891 if nr > lb
892 lb = nr * 2
893 resize!(b, lb)
894 end
895 b[nr] = a
896 end
897 if lb > olb
898 resize!(b, nr) # shrink to just contain input data if was resized
899 end
900 return nr
901 end

903 """
904 read(s::IO, nb=typemax(Int))
905
906 Read at most `nb` bytes from `s`, returning a `Vector{UInt8}` of the bytes read.
907 """
908 function read(s::IO, nb::Integer = typemax(Int))
909 # Let readbytes! grow the array progressively by default
910 # instead of taking of risk of over-allocating
911 b = Vector{UInt8}(undef, nb == typemax(Int) ? 1024 : nb)
912 nr = readbytes!(s, b, nb)
913 return resize!(b, nr)
914 end

952 ```

1003

1036 """
1037 skipchars(predicate, io::IO; linecomment=nothing)
1038
1039 Advance the stream `io` such that the next-read character will be the first remaining for
1040 which `predicate` returns `false`. If the keyword argument `linecomment` is specified, all
1041 characters from that character until the start of the next line are ignored.
1042
1043 # Examples
1044 ```jldoctest
1045 julia> buf = IOBuffer(" text")
1046 IOBuffer(data=UInt8[...], readable=true, writable=false, seekable=true, append=false, size=8, maxsize=Inf, ptr=1, mark=-1)
1047
1048 julia> skipchars(isspace, buf)
1049 IOBuffer(data=UInt8[...], readable=true, writable=false, seekable=true, append=false, size=8, maxsize=Inf, ptr=5, mark=-1)
1050
1051 julia> String(readavailable(buf))
1052 "text"
1053 ```
1054 """
1055 function skipchars(predicate, io::IO; linecomment=nothing)
1056 while !eof(io)
1057 c = read(io, Char)
1058 if c === linecomment
1059 readline(io)
1060 elseif !predicate(c)
1061 skip(io, -ncodeunits(c))
1062 break
1063 end
1064 end
1065 return io
1066 end

1068 """
1069 countlines(io::IO; eol::AbstractChar = '\\n')
1070
1071 Read `io` until the end of the stream/file and count the number of lines. To specify a file
1072 pass the filename as the first argument. EOL markers other than `'\\n'` are supported by
1073 passing them as the second argument. The last non-empty line of `io` is counted even if it does not
1074 end with the EOL, matching the length returned by [`eachline`](@ref) and [`readlines`](@ref).
1075
1076 # Examples
1077 ```jldoctest
1078 julia> io = IOBuffer("JuliaLang is a GitHub organization.\\n");
1079
1080 julia> countlines(io)
1081 1
1082
1083 julia> io = IOBuffer("JuliaLang is a GitHub organization.");
1084
1085 julia> countlines(io)
1086 1
1087
1088 julia> countlines(io, eol = '.')
1089 0
1090 ```
1091 """
1092 function countlines(io::IO; eol::AbstractChar='\n')
1093 isascii(eol) || throw(ArgumentError("only ASCII line terminators are supported"))
1094 aeol = UInt8(eol)
1095 a = Vector{UInt8}(undef, 8192)
1096 nl = nb = 0
1097 while !eof(io)
1098 nb = readbytes!(io, a)
1099 @simd for i=1:nb
1100 @inbounds nl += a[i] == aeol
1101 end
1102 end
1103 if nb > 0 && a[nb] != aeol
1104 nl += 1 # final line is not terminated with eol
1105 end
1106 nl
1107 end
